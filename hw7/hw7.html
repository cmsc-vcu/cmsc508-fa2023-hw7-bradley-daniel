List skill names of Person 1

sql = f"""
select 
    skills.name 'person_1_skills_names' 
from 
    people
    inner join peopleskills on (people.id=peopleskills.people_id)
    inner join skills on (peopleskills.skills_id=skills.id)
where 
    people.last_name='Person 1';
